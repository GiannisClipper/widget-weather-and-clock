Demo widget-weather-and-clock



User include in head section a tag like this: <script src='widget-weather-and-clock.js'></script>
and in body section one or more tags like the following examples:

<script>
new WeatherApiWidget({
latitude:40.71427,
longitude:-74.00597
}); </script>

<script>
new WeatherApiWidget({
timeRefresh:600000,
width:220,
border:'1px dashed tomato',
background:'khaki',
color:'tomato'
}); </script>

<script>
new WeatherApiWidget({
latitude:40.64361,
longitude:22.93086
}); </script>

<script>
new WeatherApiWidget({
latitude:55.753887,
longitude:37.621225
}); </script>
